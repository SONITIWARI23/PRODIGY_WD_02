<script>
  // ── State ──
  let startTime = 0;
  let elapsed = 0;
  let lapStart = 0;
  let running = false;
  let animFrame = null;
  let laps = [];

  // ── DOM ──
  const timeDisplay = document.getElementById('timeDisplay');
  const msDisplay = document.getElementById('msDisplay');
  const startBtn = document.getElementById('startBtn');
  const lapBtn = document.getElementById('lapBtn');
  const resetBtn = document.getElementById('resetBtn');
  const lapsList = document.getElementById('lapsList');
  const lapsCount = document.getElementById('lapsCount');
  const emptyState = document.getElementById('emptyState');
  const statusDot = document.getElementById('statusDot');
  const gaugeFill = document.getElementById('gaugeFill');

  const CIRCUMFERENCE = 2 * Math.PI * 110; // ~691.15

  function pad(n, d = 2) { return String(n).padStart(d, '0'); }

  function formatTime(ms) {
    const totalSec = Math.floor(ms / 1000);
    const min = Math.floor(totalSec / 60);
    const sec = totalSec % 60;
    return `${pad(min)}:${pad(sec)}`;
  }

  function formatMs(ms) {
    return '.' + pad(ms % 1000, 3);
  }

  function formatFull(ms) {
    return formatTime(ms) + formatMs(ms);
  }

  function updateGauge(ms) {
    // Rotates once per 60 seconds
    const sec = (ms / 1000) % 60;
    const progress = sec / 60;
    const offset = CIRCUMFERENCE - progress * CIRCUMFERENCE;
    gaugeFill.style.strokeDashoffset = offset;
  }

  function tick() {
    const now = performance.now();
    elapsed = now - startTime;
    const t = elapsed;

    timeDisplay.textContent = formatTime(t);
    msDisplay.textContent = formatMs(t);
    updateGauge(t);

    animFrame = requestAnimationFrame(tick);
  }

  function start() {
    startTime = performance.now() - elapsed;
    lapStart = lapStart === 0 ? startTime : (performance.now() - (elapsed - lapStart));
    running = true;

    timeDisplay.classList.add('running');
    statusDot.classList.add('running');

    startBtn.textContent = 'Pause';
    startBtn.classList.add('paused');
    lapBtn.disabled = false;
    resetBtn.disabled = false;

    animFrame = requestAnimationFrame(tick);
  }

  function pause() {
    cancelAnimationFrame(animFrame);
    running = false;

    timeDisplay.classList.remove('running');
    statusDot.classList.remove('running');

    startBtn.textContent = 'Resume';
    startBtn.classList.remove('paused');
    lapBtn.disabled = true;
  }

  function reset() {
    cancelAnimationFrame(animFrame);
    running = false;
    elapsed = 0;
    lapStart = 0;
    laps = [];

    timeDisplay.textContent = '00:00';
    timeDisplay.classList.remove('running');
    msDisplay.textContent = '.000';
    statusDot.classList.remove('running');
    gaugeFill.style.strokeDashoffset = CIRCUMFERENCE;

    startBtn.textContent = 'Start';
    startBtn.classList.remove('paused');
    lapBtn.disabled = true;
    resetBtn.disabled = true;

    lapsList.innerHTML = '';
    lapsList.appendChild(emptyState);
    lapsCount.textContent = '—';
  }

  function recordLap() {
    const now = performance.now();
    const totalElapsed = now - startTime + (elapsed - (now - startTime + elapsed - elapsed)); // = elapsed snapshot
    // simpler:
    const currentTotal = performance.now() - startTime + 0;

    // Capture current elapsed
    const lapTotalMs = performance.now() - (startTime);
    const lapSplitMs = lapStart === 0
      ? lapTotalMs
      : (performance.now() - lapStart);

    // Reset lap timer
    lapStart = performance.now();

    const lapNum = laps.length + 1;
    const lap = {
      num: lapNum,
      split: lapSplitMs,
      total: lapTotalMs
    };
    laps.push(lap);

    renderLaps();
  }

  // Patch: use simpler elapsed tracking for lap
  let lapStartTime = 0;

  function recordLapFixed() {
    const now = performance.now();
    const total = now - startTime + elapsed - elapsed; // just use running elapsed
    const totalNow = elapsed; // already updated by tick

    const split = lapStartTime === 0 ? totalNow : (totalNow - lapStartTime);
    lapStartTime = totalNow;

    const lap = {
      num: laps.length + 1,
      split: Math.max(0, split),
      total: totalNow
    };
    laps.push(lap);
    renderLaps();
  }

  function renderLaps() {
    if (laps.length === 0) {
      lapsList.innerHTML = '';
      lapsList.appendChild(emptyState);
      lapsCount.textContent = '—';
      return;
    }

    // Find best/worst split
    const splits = laps.map(l => l.split);
    const bestSplit = Math.min(...splits);
    const worstSplit = Math.max(...splits);

    lapsCount.textContent = `${laps.length} lap${laps.length > 1 ? 's' : ''}`;

    lapsList.innerHTML = '';

    // Show newest first
    [...laps].reverse().forEach((lap, i) => {
      const isBest = laps.length > 1 && lap.split === bestSplit;
      const isWorst = laps.length > 1 && lap.split === worstSplit;

      const row = document.createElement('div');
      row.className = 'lap-row' + (isBest ? ' best' : isWorst ? ' worst' : '');

      const badge = isBest
        ? '<span class="lap-badge badge-best">BEST</span>'
        : isWorst
        ? '<span class="lap-badge badge-worst">SLOW</span>'
        : '';

      row.innerHTML = `
        <span class="lap-num">L${pad(lap.num)}</span>
        <span class="lap-split">${formatFull(lap.split)}${badge}</span>
        <span class="lap-total">${formatFull(lap.total)}</span>
      `;

      lapsList.appendChild(row);
    });
  }

  // ── Override tick to track laps properly ──
  function tick2() {
    const now = performance.now();
    elapsed = now - startTime;

    timeDisplay.textContent = formatTime(elapsed);
    msDisplay.textContent = formatMs(elapsed);
    updateGauge(elapsed);

    animFrame = requestAnimationFrame(tick2);
  }

  // ── Button handlers ──
  startBtn.addEventListener('click', () => {
    if (!running) {
      if (elapsed === 0) lapStartTime = 0;
      startTime = performance.now() - elapsed;
      running = true;

      timeDisplay.classList.add('running');
      statusDot.classList.add('running');
      startBtn.textContent = 'Pause';
      lapBtn.disabled = false;
      resetBtn.disabled = false;

      animFrame = requestAnimationFrame(tick2);
    } else {
      cancelAnimationFrame(animFrame);
      running = false;

      timeDisplay.classList.remove('running');
      statusDot.classList.remove('running');
      startBtn.textContent = 'Resume';
      lapBtn.disabled = true;
    }
  });

  lapBtn.addEventListener('click', recordLapFixed);

  resetBtn.addEventListener('click', () => {
    cancelAnimationFrame(animFrame);
    running = false;
    elapsed = 0;
    lapStartTime = 0;
    laps = [];

    timeDisplay.textContent = '00:00';
    timeDisplay.classList.remove('running');
    msDisplay.textContent = '.000';
    statusDot.classList.remove('running');
    gaugeFill.style.strokeDashoffset = CIRCUMFERENCE;

    startBtn.textContent = 'Start';
    lapBtn.disabled = true;
    resetBtn.disabled = true;

    lapsList.innerHTML = '';
    lapsList.appendChild(emptyState);
    lapsCount.textContent = '—';
  });

  // Init gauge
  gaugeFill.style.strokeDasharray = CIRCUMFERENCE;
  gaugeFill.style.strokeDashoffset = CIRCUMFERENCE;
</script>
